```python
def calculate_sum(numbers):
    """Calculate the sum of a list of numbers."""
    total = 0
    for n in numbers:
        total += n
    return total

def main():
    nums = [1, 2, 3, 4, 5]
    result = calculate_sum(nums)
    print(f"Sum: {result}")
```
